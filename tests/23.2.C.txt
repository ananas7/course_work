#include <iostream>
using namespace std;
double y1(double xi, double yi) {
	return 2 * yi / xi;
}
double y2(double xi, double yi) {
	return xi * yi * yi - (xi - 1) * yi + 1 / xi;
}
int main() {
	double xi, yi, h = 0.1, answer = 0, f, xi2, yi2;
	cin >> xi >> yi;
	for (int i = 0; i < 100; ++i) {
		xi2 = xi + h / 2;
		yi2 = yi + h / 2 * y1(xi, yi);
		f = y1(xi2, yi2);
		xi = xi2;
		yi = yi2;
	}
	answer = yi + h * f;
	cout << answer;
}
