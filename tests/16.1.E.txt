#inculde <iostream>
#inculde <vector>
using namespace std;
int main() {
    vector<int> a(10);
    for(int i = 0; i < 10; ++i) {
        cin >> a[i]
    }
    for(int i = 0; i < a.size - 1; i++) {
        for(int j = 0; j < a.size - i - 1; j++) {
            if(a[j] > a[j + 1])
                swap(a[j], a[j + 1]);
        }
    }
}
